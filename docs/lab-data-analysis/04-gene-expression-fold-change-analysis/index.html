Main Python Code

def classify(fold_change: float) -> str:
    if fold_change >= 1.5:
        return "upregulated"
    if fold_change <= 0.67:
        return "downregulated"
    return "similar"


def safe_fold_change(control: float, treatment: float) -> float:
    return treatment / control if control else 0.0

Main Python Code

def gc_content(sequence: str) -> float:
    if not sequence:
        return 0.0
    gc_count = sequence.count("G") + sequence.count("C")
    return (gc_count / len(sequence)) * 100

def transcribe(sequence: str) -> str:
    return sequence.replace("T", "U")
